"""
predict.py — Single and batch inference using saved pipeline artifacts.
Usage:
    python src/predict.py
    python src/predict.py --model "Random Forest"
"""

import argparse
import pickle
import pandas as pd

ALL_FEATURES = [
    'is_franchise', 'sequel_number', 'release_year',
    'production_budget', 'marketing_budget',
    'director_avg_roi', 'lead_actor_avg_gross',
    'trailer_views_24hr', 'social_buzz_score',
    'pre_release_tracking', 'rotten_tomatoes_score',
    'genre', 'mpaa_rating', 'release_season', 'studio'
]


def load_artifacts(path: str = "models/model_artifacts.pkl") -> dict:
    with open(path, 'rb') as f:
        return pickle.load(f)


def predict(film: dict, model_name: str = "Gradient Boosting",
            artifact_path: str = "models/model_artifacts.pkl") -> float:
    """
    Predict ROI for a single film dict.
    Returns ROI as a float (e.g. 2.4 = $2.40 back per $1 spent).
    """
    arts = load_artifacts(artifact_path)
    pipe = arts['pipelines'][model_name]
    df   = pd.DataFrame([film])[ALL_FEATURES]
    return round(float(pipe.predict(df)[0]), 4)


def batch_predict(csv_path: str, model_name: str = "Gradient Boosting",
                  artifact_path: str = "models/model_artifacts.pkl",
                  output_path: str = "data/predictions.csv") -> pd.DataFrame:
    """Predict ROI for every row in a CSV and save results."""
    arts = load_artifacts(artifact_path)
    pipe = arts['pipelines'][model_name]
    df   = pd.read_csv(csv_path)
    df['predicted_roi'] = pipe.predict(df[ALL_FEATURES]).round(4)
    df.to_csv(output_path, index=False)
    print(f"✅ Predictions saved → {output_path}")
    return df


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument('--model', default='Gradient Boosting',
                        choices=['Linear Regression', 'Random Forest', 'Gradient Boosting'])
    args = parser.parse_args()

    sample = {
        'genre': 'Action', 'mpaa_rating': 'PG-13',
        'release_season': 'Summer', 'studio': 'Warner Bros',
        'is_franchise': 1, 'sequel_number': 2, 'release_year': 2024,
        'production_budget': 150_000_000, 'marketing_budget': 80_000_000,
        'director_avg_roi': 2.5, 'lead_actor_avg_gross': 200_000_000,
        'trailer_views_24hr': 12_000_000, 'social_buzz_score': 82.0,
        'pre_release_tracking': 78.0, 'rotten_tomatoes_score': 74.0,
    }

    roi = predict(sample, model_name=args.model)
    total_cost = (sample['production_budget'] + sample['marketing_budget']) / 1e6
    print(f"\nModel          : {args.model}")
    print(f"Film           : Action / Summer / PG-13 / Franchise / $150M budget")
    print(f"Predicted ROI  : {roi:.2f}×")
    print(f"Total cost     : ${total_cost:.0f}M")
    print(f"Est. revenue   : ${total_cost * (1 + roi):.1f}M")
    print(f"Verdict        : {'Profitable ✅' if roi > 1 else 'Loss ❌'}")
